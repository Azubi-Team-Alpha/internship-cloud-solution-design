### Node.js Quick Example

```javascript
const AlphaPay = require('alphapay-node');
const client = new AlphaPay(process.env.ALPHAPAY_SECRET_KEY);

const transfer = await client.transfers.create({
  amount: 500,
  currency: 'GHS',
  recipient_phone: '+233241234567',
  network: 'mtn',
  description: 'Invoice payment',
});
console.log(transfer.transfer_id);
```
